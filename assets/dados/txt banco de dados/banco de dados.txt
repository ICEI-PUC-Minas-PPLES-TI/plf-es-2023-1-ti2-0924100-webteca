CREATE TABLE leitores (ID_Leitor SMALLINT(5) NOT NULL AUTO_INCREMENT, CPF_Leitor BIGINT(11) NOT NULL, Nome_Leitor VARCHAR(45) NOT NULL, Email_Leitor VARCHAR(45) NOT NULL, Senha_Leitor VARCHAR(45) NOT NULL, Pagamento VARCHAR(45) NOT NULL, PRIMARY KEY (ID_Leitor));

CREATE TABLE avaliadores (ID_Avaliador SMALLINT(5) NOT NULL AUTO_INCREMENT, CPF_Avaliador BIGINT(11) NOT NULL, Nome_Avaliador VARCHAR(45) NOT NULL, Email_Avaliador VARCHAR(45) NOT NULL, Senha_Avaliador VARCHAR(45) NOT NULL, PRIMARY KEY (ID_Avaliador));

CREATE TABLE curriculo (ID_Curriculo SMALLINT(5) NOT NULL AUTO_INCREMENT, ID_Leitor SMALLINT(5) NOT NULL, Arquivo LONGBLOB NOT NULL, Data_Envio DATETIME NOT NULL, Data_Resposta DATETIME, Data_Dif TIME, Resposta TEXT, PRIMARY KEY (ID_Curriculo), FOREIGN KEY (ID_Leitor) REFERENCES leitores(ID_Leitor));

CREATE TABLE narradores (ID_Narrador SMALLINT(5) NOT NULL AUTO_INCREMENT, CPF_Narrador BIGINT(11) NOT NULL, Nome_Narrador VARCHAR(45) NOT NULL,
Email_Narrador VARCHAR(45) NOT NULL, Senha_Narrador VARCHAR(45) NOT NULL, ID_Curriculo SMALLINT(5) NOT NULL,
PRIMARY KEY (ID_Narrador), FOREIGN KEY (ID_Curriculo) REFERENCES curriculo(ID_Curriculo));

CREATE TABLE notificacao (ID_Notificacao SMALLINT(5) NOT NULL AUTO_INCREMENT, Descricao TEXT NOT NULL, ID_Leitor SMALLINT(5) NOT NULL,
PRIMARY KEY (ID_Notificacao), FOREIGN KEY (ID_Leitor) REFERENCES leitores(ID_Leitor));

CREATE TABLE pedido_livro (ID_PedidoLivro SMALLINT(5) NOT NULL AUTO_INCREMENT, ID_Leitor SMALLINT(5) NOT NULL, Nome_Livro VARCHAR(45) NOT NULL, Data_Envio DATETIME NOT NULL, Data_Resposta DATETIME, Data_Dif TIME, Resposta TEXT, 
PRIMARY KEY (ID_PedidoLivro), FOREIGN KEY (ID_Leitor) REFERENCES leitores(ID_Leitor));

CREATE TABLE livros (ID_Livro SMALLINT(5) NOT NULL AUTO_INCREMENT, Nome_Livro VARCHAR(45) NOT NULL,
Nome_Escritor VARCHAR(45) NOT NULL, Tema VARCHAR(45) NOT NULL, Edicao VARCHAR(45) NOT NULL, 
Idioma VARCHAR(45) NOT NULL, Audio LONGBLOB NOT NULL, Capa LONGBLOB NOT NULL, Descricao TEXT NOT NULL, 
ID_Narrador SMALLINT(5) NOT NULL,
PRIMARY KEY (ID_Livro), FOREIGN KEY (ID_Narrador) REFERENCES narradores(ID_Narrador));

CREATE TABLE pedido_remocao (ID_PedidoRemocao SMALLINT(5) NOT NULL AUTO_INCREMENT,
Nome_Livro VARCHAR(45) NOT NULL,
ID_Narrador SMALLINT(5) NOT NULL,
ID_Leitor SMALLINT(5) NOT NULL,
Motivo VARCHAR(45) NOT NULL, 
Data_Envio DATETIME NOT NULL,
Data_Resposta DATETIME, 
Data_Dif TIME, 
Resposta TEXT, 
PRIMARY KEY (ID_PedidoRemocao), 
FOREIGN KEY (ID_Leitor) REFERENCES leitores(ID_Leitor), 
FOREIGN KEY (ID_Narrador) REFERENCES narradores(ID_Narrador)); 

CREATE TABLE cartao_credito (ID_CartaoCredito SMALLINT(5) NOT NULL AUTO_INCREMENT, 
Numero_CartaoCredito VARCHAR(45) NOT NULL, Titular VARCHAR(45) NOT NULL, 
Mes_Expiracao INT NOT NULL, Ano_Expiracao INT NOT NULL, Data_Pagamento DATETIME NOT NULL,
Valor_Pagamento DECIMAL NOT NULL, ID_Leitor SMALLINT(5) NOT NULL, PRIMARY KEY (ID_CartaoCredito), 
FOREIGN KEY (ID_Leitor) REFERENCES leitores(ID_Leitor));
